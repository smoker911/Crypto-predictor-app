import requests from flask import Flask, jsonify # Initialize Flask app app = Flask(__name__) # Function to get real-time crypto data def get_crypto_data(): url = "https://api.coingecko.com/api/v3/coins/markets" params = { "vs_currency": "usd", "order": "market_cap_asc", # Small market cap coins first "per_page": 100, # Get 100 coins "page": 1 } response = requests.get(url, params=params) return response.json() # Function to detect potential 1000% gainers def detect_potential_pumps(data): potential_coins = [] for coin in data: market_cap = coin.get("market_cap", 0) price_change = coin.get("price_change_percentage_24h", 0) if market_cap and price_change: if market_cap < 50_000_000 and price_change > 100: # Small cap + big pump potential_coins.append({ "name": coin["name"], "symbol": coin["symbol"], "price": coin["current_price"], "market_cap": market_cap, "price_change_24h": price_change }) return potential_coins # API Route to get predictions @app.route('/predict', methods=['GET']) def predict(): data = get_crypto_data() pumps = detect_potential_pumps(data) return jsonify({"potential_pumps": pumps}) # Run the app if __name__ == '__main__': app.run(debug=True)
